Sàng Nguyên tố Etharonics


Tạo sẵn 1 bảng isPrime[N]
isPrime[N] = true nếu N là số nguyên tố

#include <bits/stdc++.h>
using namespace std;

int main() {
    int N = 1000000;
    vector<bool> isPrime(N + 1, true);

    isPrime[0] = isPrime[1] = false;

    for (int i = 2; i * i <= N; i++) {
        if (isPrime[i]) {
            for (int j = i * i; j <= N; j += i) {
                isPrime[j] = false;
            }
        }
    }

    // In các số nguyên tố
    for (int i = 2; i <= N; i++) {
        if (isPrime[i]) cout << i << " ";
    }
}
